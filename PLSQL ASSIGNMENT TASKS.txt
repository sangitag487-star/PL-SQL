Q7. Write a PL/SQL Program to display a given number in Reverse Order.
DECLARE
  v_num    NUMBER := &enter_number;
  v_rev    NUMBER := 0;
  v_digit  NUMBER;
  v_temp   NUMBER;
BEGIN
  v_temp := v_num;
  WHILE v_temp > 0 LOOP
    v_digit := MOD(v_temp, 10);
    v_rev   := v_rev * 10 + v_digit;
    v_temp  := TRUNC(v_temp / 10);
  END LOOP;
  DBMS_OUTPUT.PUT_LINE('Reverse of ' || v_num || ' is: ' || v_rev);
END;
/

Q13. Write a PL/SQL block which will use a user-defined procedure that accepts employee number and returns employee name and department name as OUT parameters.
CREATE OR REPLACE PROCEDURE get_emp_dept(
  p_empno   IN  NUMBER,
  p_name    OUT VARCHAR2,
  p_deptname OUT VARCHAR2
) IS
BEGIN
  SELECT e.name, d.dname
  INTO p_name, p_deptname
  FROM employee e JOIN department d ON e.deptno = d.deptno
  WHERE e.empno = p_empno;
END;
/

-- Calling block:
DECLARE
  v_name VARCHAR2(50);
  v_dept VARCHAR2(50);
BEGIN
  get_emp_dept(101, v_name, v_dept);
  DBMS_OUTPUT.PUT_LINE('Name: ' || v_name || '  Dept: ' || v_dept);
END;
/

Q14. Create a procedure which will accept Deptno and Display no of employees under different grade.
CREATE OR REPLACE PROCEDURE emp_grade_count(p_deptno NUMBER) IS
  v_a NUMBER := 0;
  v_b NUMBER := 0;
  v_c NUMBER := 0;
BEGIN
  SELECT COUNT(*) INTO v_a FROM employee
  WHERE deptno = p_deptno AND salary >= 50000;
  SELECT COUNT(*) INTO v_b FROM employee
  WHERE deptno = p_deptno AND salary >= 30000 AND salary < 50000;
  SELECT COUNT(*) INTO v_c FROM employee
  WHERE deptno = p_deptno AND salary < 30000;
  DBMS_OUTPUT.PUT_LINE('Grade A: ' || v_a);
  DBMS_OUTPUT.PUT_LINE('Grade B: ' || v_b);
  DBMS_OUTPUT.PUT_LINE('Grade C: ' || v_c);
END;
/

Q19. Create a procedure update_salary that increases the salary of a given employee by a given percentage. Parameters: employee ID, percentage.
CREATE OR REPLACE PROCEDURE update_salary(p_empno NUMBER, p_pct NUMBER) IS
BEGIN
  UPDATE employee
  SET salary = salary + (salary * p_pct / 100)
  WHERE empno = p_empno;
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Salary updated successfully.');
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('Employee not found.');
END;
/
-- Increase salary by 10%:
EXEC update_salary(101, 10);

Q20. Write a function calculate_age(p_emp_id NUMBER) that retrieves the employee's DOB, calculates age in years, and returns the age.
CREATE OR REPLACE FUNCTION calculate_age(p_emp_id NUMBER)
RETURN NUMBER IS
  v_dob  DATE;
  v_age  NUMBER;
BEGIN
  SELECT dob INTO v_dob FROM employee WHERE empno = p_emp_id;
  v_age := TRUNC(MONTHS_BETWEEN(SYSDATE, v_dob) / 12);
  RETURN v_age;
END;
/

END;

Q21. Write a PL/SQL block that will display the employee details along with salary using Cursors.
DECLARE
  CURSOR emp_cur IS
    SELECT empno, name, salary FROM employee;
  v_empno  employee.empno%TYPE;
  v_name   employee.name%TYPE;
  v_sal    employee.salary%TYPE;
BEGIN
  OPEN emp_cur;
  LOOP
    FETCH emp_cur INTO v_empno, v_name, v_sal;
    EXIT WHEN emp_cur%NOTFOUND;
    DBMS_OUTPUT.PUT_LINE(v_empno || '  ' || v_name || '  ' || v_sal);
  END LOOP;
  CLOSE emp_cur;
END;
/

Q23. Using a cursor FOR loop, increase the salary of employees in the SALES department by 10%. Also display the old salary and new salary.
DECLARE
  CURSOR sales_cur IS
    SELECT empno, name, salary FROM employee
    WHERE deptno = (SELECT deptno FROM department WHERE dname = 'SALES')
    FOR UPDATE OF salary;
  v_new_sal NUMBER;
BEGIN
  FOR rec IN sales_cur LOOP
    v_new_sal := rec.salary * 1.10;
    DBMS_OUTPUT.PUT_LINE('Emp: ' || rec.name ||
      '  Old: ' || rec.salary || '  New: ' || v_new_sal);
    UPDATE employee SET salary = v_new_sal
    WHERE CURRENT OF sales_cur;
  END LOOP;
  COMMIT;
END;
/

Q25. Given the Table Employee (EmpNo, Name, Salary, Designation, Dept_ID). Write a cursor to select the five highest paid employees from the table.
DECLARE
  CURSOR top5_cur IS
    SELECT empno, name, salary, designation, deptno
    FROM employee
    ORDER BY salary DESC
    FETCH FIRST 5 ROWS ONLY;
  v_rec top5_cur%ROWTYPE;
BEGIN
  DBMS_OUTPUT.PUT_LINE('Top 5 Highest Paid Employees:');
  DBMS_OUTPUT.PUT_LINE('--------------------------------------');
  OPEN top5_cur;
  LOOP
    FETCH top5_cur INTO v_rec;
    EXIT WHEN top5_cur%NOTFOUND;
    DBMS_OUTPUT.PUT_LINE(v_rec.name || '  Salary: ' || v_rec.salary);
  END LOOP;
  CLOSE top5_cur;
END;
/
